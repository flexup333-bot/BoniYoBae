Koneksi Google Spreadsheet Pribadi

Gunakan Google Sheets milik Anda sendiri sebagai database cloud pribadi! Tempel tautan Web App hasil Deployment Google Apps Script Anda di sini.

Jika tidak dihubungkan, semua mutasi akan disimpan otomatis di penyimpanan lokal browser Anda.

Utilitas Sinkronisasi

Gunakan tombol berikut jika spreadsheet Anda masih kosong agar format data awal langsung terisi rapi.

Panduan Sinkronisasi Database Spreadsheet

Langkah 1: Buat sebuah Spreadsheet kosong baru di Google Drive Anda.
Langkah 2: Buka Ekstensi > Apps Script. Hapus semua kode default lalu salin dan tempelkan blok kode yang sudah kami sediakan di bawah ini.
Langkah 3: Klik Terapkan > Penerapan Baru. Pilih jenis Aplikasi Web. Ubah aksesnya menjadi Siapa saja (Anyone) dan pastikan dijalankan sebagai Diri Saya (Me). Salin URL Aplikasi Web tersebut ke kotak input di atas!
Kode Google Apps Script (.gs) 
/* Kode Google Apps Script untuk Backend FinFlux - Versi Auto Header & JSONP */

function doGet(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(["ID", "Tanggal", "Kategori", "Deskripsi", "Jumlah"]);
  }
  
  var data = sheet.getDataRange().getValues();
  var result = [];
  
  for (var i = 1; i < data.length; i++) {
    result.push({
      id: data[i][0],
      tanggal: data[i][1],
      kategori: data[i][2],
      deskripsi: data[i][3],
      jumlah: Number(data[i][4])
    });
  }
  
  var jsonString = JSON.stringify(result);
  var callback = e.parameter.callback;
  if (callback) {
    return ContentService.createTextOutput(callback + '(' + jsonString + ')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  
  return ContentService.createTextOutput(jsonString)
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var params = JSON.parse(e.postData.contents);
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    
    if (sheet.getLastRow() == 0) {
      sheet.appendRow(["ID", "Tanggal", "Kategori", "Deskripsi", "Jumlah"]);
    }
    
    if (params.action === "delete") {
      var data = sheet.getDataRange().getValues();
      for (var i = 1; i < data.length; i++) {
        if (data[i][0] == params.id) {
          sheet.deleteRow(i + 1);
          break;
        }
      }
      return ContentService.createTextOutput(JSON.stringify({status: "success"}))
        .setMimeType(ContentService.MimeType.JSON);
    }
    
    sheet.appendRow([
      params.id || Utilities.getUuid(),
      params.tanggal,
      params.kategori,
      params.deskripsi,
      params.jumlah
    ]);
    
    return ContentService.createTextOutput(JSON.stringify({status: "success"}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({status: "error", message: error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
