Sinkronisasi Google Spreadsheet

Gunakan Google Sheets milik Anda sendiri sebagai database cloud pribadi! Tempel tautan Web App dari deployment Google Apps Script Anda.

Jika kosong, aplikasi akan berjalan dalam mode offline (menyimpan data di memori browser lokal).

Utilitas Spreadsheet Tambahan

Spreadsheet Anda baru dan kosong? Anda dapat mengunggah 5 data contoh lokal Anda sekarang agar langsung terisi rapi dengan strukturnya.

Panduan 3 Langkah Penyusunan Database Spreadsheet Anda

Langkah 1: Buat sebuah Spreadsheet baru di Google Drive Anda. Berikan nama yang sesuai, contoh: Mutasi Finansial FinFlux.
Langkah 2: Klik Ekstensi > Apps Script. Hapus semua kode default dan tempelkan blok kode yang sudah kami sediakan di bawah ini.
Langkah 3: Klik tombol Terapkan > Penerapan Baru. Pilih jenis Aplikasi Web. Ubah aksesnya menjadi Siapa saja (Anyone) dan pastikan dijalankan sebagai Diri Anda (Me). Salin URL Aplikasi Web yang dihasilkan dan masukkan ke kotak input di atas!

/* Kode Google Apps Script untuk Backend FinFlux - Mendukung Auto Header */

function doGet(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  
  // Jika Spreadsheet masih kosong (tidak ada baris)
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(["ID", "Tanggal", "Kategori", "Deskripsi", "Jumlah"]);
  }
  
  var data = sheet.getDataRange().getValues();
  var result = [];
  
  // Ambil data jika ada selain header
  for (var i = 1; i < data.length; i++) {
    result.push({
      id: data[i][0],
      tanggal: data[i][1],
      kategori: data[i][2],
      deskripsi: data[i][3],
      jumlah: Number(data[i][4])
    });
  }
  
  var jsonString = JSON.stringify(result);
  
  // Deteksi callback untuk penanganan JSONP (Bypass pembatasan CORS peramban)
  var callback = e.parameter.callback;
  if (callback) {
    return ContentService.createTextOutput(callback + '(' + jsonString + ')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  
  return ContentService.createTextOutput(jsonString)
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var params = JSON.parse(e.postData.contents);
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    
    // Header setup otomatis jika sheet kosong
    if (sheet.getLastRow() == 0) {
      sheet.appendRow(["ID", "Tanggal", "Kategori", "Deskripsi", "Jumlah"]);
    }
    
    if (params.action === "delete") {
      var data = sheet.getDataRange().getValues();
      for (var i = 1; i < data.length; i++) {
        if (data[i][0] == params.id) {
          sheet.deleteRow(i + 1);
          break;
        }
      }
      return ContentService.createTextOutput(JSON.stringify({status: "success"}))
        .setMimeType(ContentService.MimeType.JSON);
    }
    
    // Tambah baris baru
    sheet.appendRow([
      params.id || Utilities.getUuid(),
      params.tanggal,
      params.kategori,
      params.deskripsi,
      params.jumlah
    ]);
    
    return ContentService.createTextOutput(JSON.stringify({status: "success"}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({status: "error", message: error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
